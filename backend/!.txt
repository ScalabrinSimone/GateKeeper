import asyncio
from bleak import BleakScanner
import datetime
import sqlite3

# Configurazione database
def save_device(address, name):
    conn = sqlite3.connect('access_log.db')
    c = conn.cursor()
    c.execute("INSERT INTO logs (address, name, timestamp) VALUES (?, ?, ?)", 
              (address, name, datetime.datetime.now()))
    conn.commit()
    conn.close()

def detection_callback(device, advertisement_data):
    # Filtra solo i dispositivi che ti interessano (es. per nome o indirizzo)
    if device.name:
        print(f"Rilevato: {device.name} [{device.address}]")
        save_device(device.address, device.name)

async def main():
    scanner = BleakScanner(detection_callback)
    await scanner.start()
    await asyncio.sleep(float('inf')) # Mantiene lo scanner attivo indefinitamente

asyncio.run(main())
